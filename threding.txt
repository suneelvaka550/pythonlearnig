import threading
import time

def squres(n):
    for i in range(1,n+1):
        print(i*i ,end=' ')
def qubes(n):
    for i in range(1,n+1):
        print("quebes values is :",i*i*i)
 
intial=time.time()       
print("startingtime:",intial)
t1=threading.Thread(target=squres,args=(4,))
t2=threading.Thread(target=qubes,args=(4,))

t1.start()
t2.start()
endtime=time.time()
dif=endtime-intial
print("running time is :",dif)
